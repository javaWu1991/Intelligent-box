	@RequestMapping(value = "/userinfo/{account:\\d+}/{cid:\\d+}", method = RequestMethod.GET)
	@ResponseBody
	public JsonResult userinfo(@PathVariable String account, @PathVariable Long cid) {
		TAppUser user = appUserService.selectByAccount(account);
		if (null != user) {
			user.setPassword(null);
			Map<String, Object> map = new HashMap<String, Object>();
			map.put("uid", user.getId());
			List<Long> list = new ArrayList<>();
			list.add(cid);
			map.put("list", list);
			List<TAppOrganization> orts = organizationMapper.selectAllByCidsUid(map);
			if (CollectionUtils.isNotEmpty(orts)) {
				user.setOrgId(orts.get(0).getId());
				user.setOrgName(orts.get(0).getOrgName());
			}
			return new JsonResult(true, "", user);
		}
		return new JsonResult(false, "用户信息不存在！", null);
	}

	@RequestMapping("/getUserHead/{id}")
	public void getUserHeadImg(@PathVariable Long id, HttpServletResponse response) throws IOException {
		TAppUser tAppUser = appUserService.selectByPrimaryKey(id);
		String url = tAppUser.getHeadUrl();

		if (StringUtils.isEmpty(url)) {
			response.setContentType("application/json");
		} else {
			FileNameMap fileNameMap = URLConnection.getFileNameMap();
			String type = fileNameMap.getContentTypeFor(url);
			response.setContentType(type);
			FileInputStream is = new FileInputStream(FileUpload.OUT_FILE_PATH + url);
			byte[] buffer = new byte[is.available()];
			is.read(buffer);
			ServletOutputStream out = response.getOutputStream();
			out.write(buffer);
			out.flush();
			out.close();
			is.close();
		}
	}

	@RequestMapping("/getUserHeadForAndroid/{id}")
	public void getUserImg(@PathVariable Long id, HttpServletResponse response, Boolean isCompress) throws IOException {
		TAppUser tAppUser = appUserService.selectByPrimaryKey(id);
		String url = tAppUser.getHeadUrl();

		if (StringUtils.isEmpty(url)) {
			url = "/upload/headImage/" + id % 5 + ".png";
		}

		FileNameMap fileNameMap = URLConnection.getFileNameMap();
		String type = fileNameMap.getContentTypeFor(url);
		response.setContentType(type);
		FileInputStream is;
		if (isCompress == null) {
			String iString = url.substring(0, url.lastIndexOf("/")) + "/compress" + url.substring(url.lastIndexOf("/"));
			is = new FileInputStream(FileUpload.OUT_FILE_PATH + iString);
		} else {
			is = new FileInputStream(FileUpload.OUT_FILE_PATH + url);
		}
		byte[] buffer = new byte[is.available()];
		is.read(buffer);
		ServletOutputStream out = response.getOutputStream();
		out.write(buffer);
		out.flush();
		out.close();
		is.close();

	}

	/**
	 * 选择集团登陆之后保存APP端的ClientId
	 * 
	 * @param account
	 * @param client
	 * @param type
	 * @return
	 */
	@RequestMapping(value = "/setClient/{cid:\\d+}/{account:\\d+}/{client}/{type:\\d+}", method = RequestMethod.GET)
	@ResponseBody
	public JsonResult setClient(@PathVariable("cid") String cid, @PathVariable("account") String account, @PathVariable("client") String client,
			@PathVariable("type") String type, String deviceToken) {
		if (StringUtils.isNotEmpty(cid) && StringUtils.isNotEmpty(account) && StringUtils.isNotEmpty(client) && StringUtils.isNotEmpty(type)) {
			Map<String, Object> map = new HashMap<String, Object>();
			map.put("account", account);
			map.put("cid", Long.parseLong(cid));
			map.put("clientId", client);
			map.put("token", deviceToken);
			map.put("type", type);
			List<TAppUserClient> list = userClientMapper.selectByParams(map);
			if (CollectionUtils.isEmpty(list)) {
				TAppUserClient userClient = new TAppUserClient();
				userClient.setAccount(account);
				userClient.setCid(Long.parseLong(cid));
				userClient.setClientId(client);
				userClient.setDeviceToken(deviceToken);
				userClient.setPhoneType(Integer.parseInt(type));
				userClientMapper.insertSelective(userClient);
			}
			return new JsonResult(true, null, null);
		}
		return new JsonResult(false, "参数错误！", null);
	}

	/**
	 * 修改密码
	 * 
	 * @param vo
	 * @param smsTime
	 * @param code
	 * @param valiCode
	 * @return
	 */
	@RequestMapping(value = "/updatePass", method = RequestMethod.POST)
	@ResponseBody
	public JsonResult updatePass(AppUserVo vo, String smsTime, String code, String valiCode) {
		JsonResult result = checkValiCode(vo.getAccount(), smsTime, code, valiCode);
		if (result.getSuccess()) {

			if (StringUtils.isEmpty(vo.getPassword())) {
				return new JsonResult(false, "密码参数缺失！", null);
			}

			TAppUser user = appUserService.selectByAccount(vo.getAccount());
			if (null != user) {
				if (user.getStatus() == 0) {
					return new JsonResult(false, "该账号已停用，请联系管理员！", null);
				} else if (user.getStatus() == Constants.USER_NOT_REG_IM) {
					return new JsonResult(false, "非法用户！", null);
				}
				/**
				 * 密码未修改
				 */
				if (user.getPassword().equals(vo.getPassword())) {
					return new JsonResult(false, "请填写您近期未使用的密码！", null);
				}

				TAppUser userVo = new TAppUser();
				userVo.setId(user.getId());
				userVo.setPassword(vo.getPassword());
				int success = appUserService.updatePasswordByApp(userVo);
				if (success == 1) {
					return new JsonResult(true, "修改成功", null);
				} else {
					return new JsonResult(false, "修改失败！", null);
				}
			}
		}
		return result;
	}

	/**
	 * 登陆接口
	 * 
	 * @return
	 */
	@RequestMapping(value = "/login", method = RequestMethod.POST)
	@ResponseBody
	public JsonResult apiLogin(String account, String password, String type) {
		if (StringUtils.isNotEmpty(account) && StringUtils.isNotEmpty(password)) {
			if (StringUtils.isEmpty(type)) {
				return new JsonResult(false, "登陆设备类型不明！！", null);
			}

			if (!CheckoutUtil.checkMobile(account)) {
				return new JsonResult(false, "错误的手机号码！", null);
			}

			/**
			 * 校验手机号码是否注册过
			 */
			TAppUser user = appUserService.selectByAccount(account);
			if (null == user) {
				return new JsonResult(false, "该号码未注册！请先注册！", null);
			}
			if (RandomUtil.MD5(password).equalsIgnoreCase(user.getPassword())) {
				if (user.getStatus() == 0) {
					return new JsonResult(false, "该账号已停用，请联系管理员！", null);
				} else if (user.getStatus() == Constants.USER_NOT_REG_IM) {
					// 注册小溪并登陆
					JsonResult result = appUserService.apiRegisterIm(user.getName(), user.getAccount(), user.getPassword(), user.getSex(), false);
					if (result.getSuccess()) {
						user.setStatus(Constants.USER_CAN_USE);
						user.setPassword(null);
						appUserService.updateByPrimaryKeySelective(user);
						// return result;
					} else {
						return new JsonResult(false, "注册小溪失败！请联系管理员！", null);
					}
				}

				AppUserVo appUserVo = new AppUserVo();
				BeanUtils.copyProperties(user, appUserVo);
				appUserVo.setPassword("******");

				List<TAppCompany> list = appUserService.findCompanyByUser(user.getId(), user.getAccount(), true);
				if (CollectionUtils.isNotEmpty(list)) {
					if (user.getStatus() == Constants.USER_NOT_VALIDATE) {
						return new JsonResult(false, "请等待审核通过！", null);
					}
					appUserVo.setCompanys(list);
					TAppUpdateLog log = updateLogMapper.selectNewVersionByVersion(Integer.parseInt(type));
					if (null != log) {
						appUserVo.setDownloadUrl(log.getFilePath());
					}
					return new JsonResult(true, "", appUserVo);
				} else {
					// 没有加入公司的APP端允许其登录
					return new JsonResult(false, "请申请加入企业或创建企业！", appUserVo);
				}
			} else {
				return new JsonResult(false, "用户名或密码错误！", null);
			}
		}
		return new JsonResult(false, "请输入用户名或密码", null);
	}

	/**
	 * 注册接口
	 * 
	 * @return
	 */
	@RequestMapping(value = "/register", method = RequestMethod.POST)
	@ResponseBody
	public JsonResult apiRegister(AppUserVo userVo, String smsTime, String code, String valiCode) {
		if (!CheckoutUtil.checkMobile(userVo.getMobile())) {
			return new JsonResult(false, "错误的手机号码！", null);
		}
		JsonResult result = checkValiCode(userVo.getMobile(), smsTime, code, valiCode);
		if (result.getSuccess()) {
			String mobile = userVo.getMobile();
			String pass = userVo.getPassword();
			if (StringUtils.isNotEmpty(mobile) && StringUtils.isNotEmpty(pass)) {
				result = appUserService.apiRegister(userVo.getName(), mobile, pass, userVo.getSex(), false);
				TAppUser appUser = (TAppUser) result.getModel();
				if (!result.getSuccess()) {
					AppUserVo appUserVo = new AppUserVo();
					BeanUtils.copyProperties(appUser, appUserVo);
					appUserVo.setPassword("******");
					List<TAppCompany> list = appUserService.findCompanyByUser(appUser.getId(), appUser.getAccount(), true);
					if (CollectionUtils.isEmpty(list)) {
						return new JsonResult(false, "请申请加入企业或创建企业！", appUserVo);
					}
				}
				return result;
			}
			return new JsonResult(false, "手机号或密码参数缺失！", null);
		}
		return result;
	}

	 */
	@RequestMapping(value = "/applyJoinCompany", method = RequestMethod.POST)
	@ResponseBody
	public JsonResult applyJoinCompany(String account, String name, String email, String shortNum, String jobId, String cid, String orgId) {

		if (StringUtils.isEmpty(account)) {
			return new JsonResult(false, "手机号码缺失！", null);
		}

		if (!CheckoutUtil.checkMobile(account)) {
			return new JsonResult(false, "错误的手机号码！", null);
		}

		if (StringUtils.isEmpty(jobId)) {
			return new JsonResult(false, "职位信息缺失！", null);
		}
		if (StringUtils.isEmpty(cid)) {
			return new JsonResult(false, "企业信息缺失！", null);
		}
		if (StringUtils.isEmpty(orgId)) {
			return new JsonResult(false, "部门信息缺失！", null);
		}
		if (StringUtils.isEmpty(name)) {
			return new JsonResult(false, "姓名信息缺失！", null);
		}

		if (StringUtils.isNotEmpty(email)) {
			if (!CheckoutUtil.checkEmail(email)) {
				return new JsonResult(false, "邮箱格式不合法！", null);
			}
		}
		if (StringUtils.isNoneEmpty(shortNum)) {
			Boolean isNum = shortNum.matches("[0-9]+");
			if (!isNum) {
				return new JsonResult(false, "短号格式不合法！", null);
			}
		}

		TAppUser user = appUserService.selectByAccount(account);
		if (null != user) {
			TAppUserOrg org = userOrgMapper.selectByOidUid(Long.parseLong(orgId), user.getId());
			if (null == org) {
				TAppUserOrg userOrg = new TAppUserOrg();
				userOrg.setOid(Long.parseLong(orgId));
				userOrg.setStatus(Constants.USER_NOT_VALIDATE);
				userOrg.setUid(user.getId());
				userOrg.setPositionId(Integer.valueOf(jobId));
				userOrg.setEmail(email);
				userOrg.setShortNum(shortNum);
				userOrgMapper.insertSelective(userOrg);
				return new JsonResult(true, "申请加入企业成功！", null);
			} else if (org.getStatus() == 1) {
				return new JsonResult(false, "您已加入该部门，不可再次申请！", null);
			} else if (org.getStatus() == -2) {
				return new JsonResult(false, "您已申请加入该部门，请等待管理员审核！", null);
			}
		}
		return new JsonResult(false, "手机号码未注册！", null);
	}

	 * 
	 * @return
	 */
	@RequestMapping(value = "/approval/{cid}", method = RequestMethod.GET)
	@ResponseBody
	public JsonResult findApprovalUser(@PathVariable String cid) {

		List<TAppUser> users = appUserService.selectPendingApproval(cid);
		if (users != null && !users.isEmpty()) {
			return new JsonResult(true, "获取待审核用户成功", users);
		}
		return new JsonResult(false, "获取待审核用户失败", users);
	}

	/**
	 * 更新待审核用户信息接口
	 * 
	 * @return
	 */
	@RequestMapping(value = "/approval/update", method = RequestMethod.POST)
	@ResponseBody
	public JsonResult updateApprovalUser(TAppUser user, Long orgId, Long cid, Long positionId) {

		try {
			user.setUpdateTime(System.currentTimeMillis());
			user.setPassword(null);
			appUserService.updateByPrimaryKeySelective(user);
			if (positionId != null || user.getShortNum() != null) {
				System.out.println("positionId【" + positionId + "】、shortNum【" + user.getShortNum() + "】 等待更新...");
				userOrgMapper.updateUserOrg(user.getId(), orgId, cid, positionId, user.getShortNum());
			}
			return new JsonResult(true, "更新待审核用户成功", null);
		} catch (Exception e) {
			return new JsonResult(false, "更新待审核用户失败,失败原因：" + e.getMessage(), null);
		}
	}
